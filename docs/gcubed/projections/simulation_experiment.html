from genericpath import isfile
import logging
import os
import os.path
from numpy import empty
import pandas as pd
from gcubed.base import Base
from gcubed.projections.projections import Projections
from gcubed.projections.baseline_projections import BaselineProjections
from gcubed.projections.simulation_layer_definitions import SimulationLayerDefinitions
from gcubed.projections.simulation_layer import SimulationLayer

class SimulationExperiment(Base):

    """
    Sets up a standard simulation experiment using the provided baseline projections and
    the simulation layers described in the experiment's design file.
    """

    def __init__(self, baseline_projections: BaselineProjections, design_file: str = None) -> None:
        """
        Initialises the experiment design by loading a design CSV file.
        
       ### Arguments
            baseline_projections: The baseline projection that the simulations will be applied to.
            design_file: the name and relative path to the experiment configuration file from the simulations directory,
            e.g. 'example_experiment/design.csv'
        """

        assert baseline_projections is not None
        self._baseline_projections = baseline_projections

        assert design_file is not None
        filename: str = f"{self.baseline_projections.configuration.simulations_directory}{design_file}"
        if not os.path.isfile(filename):
            raise Exception(f"The experiment design file {filename} does not exist.")

        self._simulation_layer_definitions = SimulationLayerDefinitions(sym_data=self.baseline_projections.sym_data)
        self.simulation_layer_definitions.load_from_csv_file(design_file=design_file)

        self.__run_simulations()

        self.validate()

    def validate(self):
        """
        TODO: Validate the experiment, checking that it will work with the given baseline
        and checking that the simulation layers make sense.
        """
        pass

    def __run_simulations(self):
        """
        Create the simulation layers, one for each definition.
        """
        self._simulation_layers: list[SimulationLayer] = []
        previous_projections: Projections = self.baseline_projections
        for simulation_layer_definition in self.simulation_layer_definitions:
            self.simulation_layers.append(
                SimulationLayer(
                    simulation_layer_definition=simulation_layer_definition,
                    previous_projections=previous_projections
                )
            )
            previous_projections = self.simulation_layers[-1]

    @property
    def baseline_projections(self) -> BaselineProjections:
        return self._baseline_projections

    @property
    def simulation_layer_definitions(self) -> SimulationLayerDefinitions:
        """
        Returns the list of simulation layer definitions, in their appropriate order
        from the first to be applied to the last to be applied. 
        """
        return self._simulation_layer_definitions

    @property
    def simulation_layers(self) -> list[SimulationLayer]:
        return self._simulation_layers

    @property
    def final_projections(self) -> Projections:
        """
        Return the final projections generated by the experiment.
        """
        if self.simulation_layers is None or len(self.simulation_layers) == 0:
            return self.baseline_projections
        return self.simulation_layers[-1]
